# 数据库设计

## 1.driver

```sql
CREATE TABLE `driver` (
  `id` varchar(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `name` varchar(255) NOT NULL,
  `wechat` varchar(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci DEFAULT NULL,
  `punishLog` json DEFAULT NULL COMMENT '罚款日志',
  `drivingLog` json DEFAULT NULL COMMENT '行车日志',
  `repairReportLog` json DEFAULT NULL COMMENT '维修申报日志',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
```

## 2.manager

```sql
CREATE TABLE `manager` (
  `id` varchar(255) NOT NULL,
  `name` varchar(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `wechat` varchar(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci DEFAULT NULL,
  `repairReportLog` json DEFAULT NULL COMMENT '维修申报日志',
  `assignReportLog` json DEFAULT NULL COMMENT '派车申报日志',
  `lendReportLog` json DEFAULT NULL COMMENT '借出申报日志',
  `protectLog` json DEFAULT NULL COMMENT '保养日志',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
```

## 3.officer

```sql
CREATE TABLE `officer` (
  `id` varchar(255) NOT NULL,
  `name` varchar(255) NOT NULL,
  `wechat` varchar(255) DEFAULT NULL,
  `drivingAgreeLog` json DEFAULT NULL COMMENT '派车审批日志',
  `repairAgreeLog` json DEFAULT NULL COMMENT '维修审批日志',
  `lendAgreeLog` json DEFAULT NULL COMMENT '出借审批日志',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
```

## 4.vehicle

```sql
CREATE TABLE `vehicle` (
  `license` varchar(255) NOT NULL COMMENT '车牌号',
  `status` int NOT NULL COMMENT '车辆状态 0代表待用 1代表出行 2代表维修',
  `workingLog` json DEFAULT NULL COMMENT '车辆出行日志',
  `protectLog` json DEFAULT NULL COMMENT '保养日志',
  `repairLog` json DEFAULT NULL COMMENT '维修日志',
  `violationLog` json DEFAULT NULL COMMENT '违规日志',
  PRIMARY KEY (`license`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
```
